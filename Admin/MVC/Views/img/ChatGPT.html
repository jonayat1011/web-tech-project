ChatGPT

Here's the complete HTML code incorporating the Chart.js library and the pie chart setup:

html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Appointment Pie Chart</title>
    <!-- Link to Chart.js library -->
    <script src="https://cdn.jsdelivr.net/npm/chart.js"></script>
</head>
<body>
    <div class="box appointmentChart">
        <!-- Content for Appointment Pie Chart -->
        <h2>Appointment Pie Chart</h2>
        <div class="PieChart">
            <!-- Canvas for the pie chart -->
            <canvas id="appointmentPieChart" width="400" height="400"></canvas>
        </div>
    </div>

    <script>
        // Get the canvas element
        var ctx = document.getElementById('appointmentPieChart').getContext('2d');

        // Data for the pie chart
        var confirmedAppointment = 500;
        var canceledAppointment = 120;
        var data = {
            datasets: [{
                data: [confirmedAppointment, canceledAppointment],
                backgroundColor: [
                    'rgba(75, 192, 192, 0.2)', // Color for confirmed appointments
                    'rgba(255, 99, 132, 0.2)' // Color for canceled appointments
                ],
                borderColor: [
                    'rgba(75, 192, 192, 1)', // Border color for confirmed appointments
                    'rgba(255, 99, 132, 1)' // Border color for canceled appointments
                ],
                borderWidth: 1
            }],

            // These labels appear in the legend and in the tooltips when hovering different arcs
            labels: [
                'Confirmed Appointments',
                'Canceled Appointments'
            ]
        };

        // Configuration options
        var options = {
            responsive: true,
            maintainAspectRatio: false,
        };

        // Initialize the pie chart
        var appointmentPieChart = new Chart(ctx, {
            type: 'pie',
            data: data,
            options: options
        });
    </script>
</body>
</html>

            type: 'pie',
            data: data,
            options: options

            type: 'pie',
            data: data,
            options: options
        });
